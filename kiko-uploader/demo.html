🔧 Kiko API

Access and control the component through JavaScript

Attributes

show-filename

Show/hide filename display on thumbnails. Default: false

show-filename="true"

show-filesize

Show/hide file size display on thumbnails. Default: false

show-filesize="true"

show-file-list

Show/hide the detailed file list table. Default: false

show-file-list="true"

uploader-bg-color

Background color of the upload area. Default: #fafafa

uploader-bg-color="#f0f8ff"

uploader-border-color

Border color of the upload area. Default: #ccc

uploader-border-color="#4285f4"

max-desc-length

Maximum character limit for file descriptions. 0 = unlimited. Default: 0

max-desc-length="200"

single-folder-mode

Allow files in only one top-level folder at a time. Default: false

single-folder-mode="true"

single-folder-hint

Custom hint text for single folder mode.

single-folder-hint="Custom hint message"

groups

JSON string defining folder structure with nested groups.

groups='[{"id":"g1","name":"Group 1","has-file":0,"groups":[...]}]'

default-expand

Whether groups should be expanded by default. Default: true

default-expand="false"

Methods

getUploadedFiles()

Returns an array of uploaded files with their metadata.

uploader.getUploadedFiles() → Array<{file, order, group, groupItemOrder, thumbnail, description}>

setUploadedFiles(files)

Programmatically set files in the uploader.

uploader.setUploadedFiles([{file, group?, thumbnail?, description?}])

setGroups(groups)

Set the groups configuration programmatically.

uploader.setGroups([{id, name, order?, width?, "has-file"?, groups?}])

addGroup(name, id?)

Add a new group for organizing files.

uploader.addGroup("Group Name") → groupId

removeGroup(groupId)

Remove a group (files move to ungrouped).

uploader.removeGroup("groupId")

clearAllFiles()

Remove all uploaded files.

uploader.clearAllFiles()

Group Configuration

Groups support hierarchical organization with the following properties:

Basic Group

{"id": "unique-id", "name": "Display Name", "order": 0}

Container Group (no files)

{"id": "container", "name": "Container", "has-file": 0, "groups": [...]}

Custom Width Group

{"id": "wide", "name": "Wide Group", "width": "50%"}

Default Expand Control

{"id": "collapsed", "name": "Collapsed Group", "default-expand": false}

Per-Group Single Folder Mode

{"id": "single", "name": "Single Mode Group", "single-folder-mode": "true", "single-folder-hint": "Custom hint"}

Nested Groups Example

{"id": "projects", "name": "Projects", "has-file": 0, "groups": [ {"id": "web", "name": "Web Dev", "width": "50%"}, {"id": "mobile", "name": "Mobile", "width": "50%"} ]}

Advanced Group Example

{"id": "advanced", "name": "Advanced Group", "order": 0, "width": "100%", "has-file": 1, "default-expand": true, "single-folder-mode": "false", "groups": []}

Events

file-added

Fired when a file is uploaded. Detail: {file, index}

file-removed

Fired when a file is removed. Detail: {index}

file-moved

Fired when a file is moved between groups. Detail: {fileIndex, groupId, fromGroupId}

description-updated

Fired when a file description is updated. Detail: {fileIndex, description}

files-cleared

Fired when all files are cleared.

files-set

Fired when files are set programmatically. Detail: {count}

React Component

import React, { useRef, useEffect } from 'react'; import './kiko-uploader.min.js'; function KikoUploaderApp() { const uploaderRef = useRef(null); useEffect(() => { const uploader = uploaderRef.current; const handleFileAdded = (event) => { console.log('File added:', event.detail); }; uploader.addEventListener('file-added', handleFileAdded); return () => { uploader.removeEventListener('file-added', handleFileAdded); }; }, []); const getFiles = () => { const files = uploaderRef.current.getUploadedFiles(); console.log('Uploaded files:', files); }; const addGroup = () => { uploaderRef.current.addGroup('New Group'); }; const clearFiles = () => { uploaderRef.current.clearAllFiles(); }; const setGroups = () => { uploaderRef.current.setGroups([ {id: 'docs', name: 'Documents', order: 0}, {id: 'images', name: 'Images', order: 1} ]); }; const setAdvancedGroups = () => { uploaderRef.current.setGroups([ { id: 'projects', name: 'Projects', order: 0, hasFile: 0, groups: [ { id: 'web', name: 'Web Development', order: 0, width: '50%' }, { id: 'mobile', name: 'Mobile Apps', order: 1, width: '50%' } ] }, { id: 'documents', name: 'Documents', order: 1, width: '100%' } ]); }; return ( <div> <kiko-uploader ref={uploaderRef} show-filename="true" show-filesize="true" uploader-bg-color="#f0f8ff" uploader-border-color="#4285f4" max-desc-length="300" groups='[{"id":"docs","name":"Documents"}]' /> <div style={{ marginTop: '10px' }}> <button onClick={getFiles} style={{ marginRight: '10px' }}>Get Files</button> <button onClick={addGroup} style={{ marginRight: '10px' }}>Add Group</button> <button onClick={setAdvancedGroups} style={{ marginRight: '10px' }}>Set Advanced Groups</button> <button onClick={clearFiles}>Clear Files</button> </div> </div> ); } export default KikoUploaderApp;

React with Redux

Integrate the file uploader with Redux for centralized state management.

Redux Store Setup

// store.js import { createStore } from 'redux'; import kikoUploaderReducer from './reducers/kikoUploaderReducer'; const store = createStore(kikoUploaderReducer); export default store;

Actions

// actions/kikoUploaderActions.js export const ADD_FILE = 'ADD_FILE'; export const REMOVE_FILE = 'REMOVE_FILE'; export const SET_FILES = 'SET_FILES'; export const UPDATE_DESCRIPTION = 'UPDATE_DESCRIPTION'; export const ADD_GROUP = 'ADD_GROUP'; export const CLEAR_FILES = 'CLEAR_FILES'; export const addFile = (fileData) => ({ type: ADD_FILE, payload: fileData }); export const removeFile = (fileIndex) => ({ type: REMOVE_FILE, payload: fileIndex }); export const setFiles = (files) => ({ type: SET_FILES, payload: files }); export const updateDescription = (fileIndex, description) => ({ type: UPDATE_DESCRIPTION, payload: { fileIndex, description } }); export const addGroup = (groupName) => ({ type: ADD_GROUP, payload: groupName }); export const clearFiles = () => ({ type: CLEAR_FILES });

Reducer

// reducers/kikoUploaderReducer.js import { ADD_FILE, REMOVE_FILE, SET_FILES, UPDATE_DESCRIPTION, ADD_GROUP, CLEAR_FILES } from '../actions/kikoUploaderActions'; const initialState = { files: [], groups: [ { id: 'reports', name: 'Annual Reports', order: 0 }, { id: 'documents', name: 'General documents', order: 1 }, { id: 'images', name: 'Images', order: 2 } ] }; const kikoUploaderReducer = (state = initialState, action) => { switch (action.type) { case ADD_FILE: return { ...state, files: [...state.files, action.payload] }; case REMOVE_FILE: return { ...state, files: state.files.filter((_, index) => index !== action.payload) }; case SET_FILES: return { ...state, files: action.payload }; case UPDATE_DESCRIPTION: return { ...state, files: state.files.map((file, index) => index === action.payload.fileIndex ? { ...file, description: action.payload.description } : file ) }; case ADD_GROUP: const newGroupId = `group_${Date.now()}`; return { ...state, groups: [...state.groups, { id: newGroupId, name: action.payload, order: state.groups.length }] }; case CLEAR_FILES: return { ...state, files: [] }; default: return state; } }; export default kikoUploaderReducer;

React Component with Redux

// components/KikoUploaderRedux.js import React, { useRef, useEffect } from 'react'; import { useSelector, useDispatch } from 'react-redux'; import { addFile, removeFile, setFiles, updateDescription, addGroup, clearFiles } from '../actions/kikoUploaderActions'; import './kiko-uploader.min.js'; function KikoUploaderRedux() { const uploaderRef = useRef(null); const dispatch = useDispatch(); const { files, groups } = useSelector(state => state); useEffect(() => { const uploader = uploaderRef.current; const handleFileAdded = (event) => { dispatch(addFile(event.detail)); }; const handleFileRemoved = (event) => { dispatch(removeFile(event.detail.index)); }; const handleDescriptionUpdated = (event) => { dispatch(updateDescription(event.detail.fileIndex, event.detail.description)); }; const handleFilesCleared = () => { dispatch(clearFiles()); }; uploader.addEventListener('file-added', handleFileAdded); uploader.addEventListener('file-removed', handleFileRemoved); uploader.addEventListener('description-updated', handleDescriptionUpdated); uploader.addEventListener('files-cleared', handleFilesCleared); return () => { uploader.removeEventListener('file-added', handleFileAdded); uploader.removeEventListener('file-removed', handleFileRemoved); uploader.removeEventListener('description-updated', handleDescriptionUpdated); uploader.removeEventListener('files-cleared', handleFilesCleared); }; }, [dispatch]); const handleGetFiles = () => { const uploaderFiles = uploaderRef.current.getUploadedFiles(); dispatch(setFiles(uploaderFiles)); console.log('Files from Redux state:', files); }; const handleAddGroup = () => { const groupName = 'New Redux Group'; dispatch(addGroup(groupName)); uploaderRef.current.addGroup(groupName); }; const handleClearFiles = () => { uploaderRef.current.clearAllFiles(); }; return ( <div> <h3>Redux State</h3> <p>Files in Redux: {files.length}</p> <p>Groups in Redux: {groups.length}</p> <div style={{ marginBottom: '20px' }}> <button onClick={handleGetFiles} style={{ marginRight: '10px' }}> Sync Files to Redux </button> <button onClick={handleAddGroup} style={{ marginRight: '10px' }}> Add Group </button> <button onClick={handleClearFiles}> Clear All Files </button> </div> <kiko-uploader ref={uploaderRef} show-filename="true" show-filesize="true" uploader-bg-color="#e8f5e8" uploader-border-color="#4caf50" groups={JSON.stringify(groups)} /> </div> ); } export default KikoUploaderRedux;

App Setup

// App.js import React from 'react'; import { Provider } from 'react-redux'; import store from './store'; import KikoUploaderRedux from './components/KikoUploaderRedux'; function App() { return ( <Provider store={store}> <div className="App"> <h1>Kiko Uploader with Redux</h1> <KikoUploaderRedux /> </div> </Provider> ); } export default App;
