🔧 Kiko API

Access and control the component through JavaScript

Methods

getUploadedFiles()

Returns an array of uploaded files with their metadata.

uploader.getUploadedFiles() → Array<{file, order, group, groupItemOrder, thumbnail, description}>

setUploadedFiles(files)

Programmatically set files in the uploader.

uploader.setUploadedFiles([{file, group?, thumbnail?, description?}])

addGroup(name, id?)

Add a new group for organizing files.

uploader.addGroup("Group Name") → groupId

removeGroup(groupId)

Remove a group (files move to ungrouped).

uploader.removeGroup("groupId")

clearAllFiles()

Remove all uploaded files.

uploader.clearAllFiles()

Events

file-added

Fired when a file is uploaded. Detail: {file, index}

file-removed

Fired when a file is removed. Detail: {index}

file-moved

Fired when a file is moved between groups. Detail: {fileIndex, groupId, fromGroupId}

description-updated

Fired when a file description is updated. Detail: {fileIndex, description}

files-cleared

Fired when all files are cleared.

files-set

Fired when files are set programmatically. Detail: {count}

Let's see how to use Kiko Uploader in different frameworks

Plain HTML

Simply include the component script and use the custom element in your HTML.

HTML

<!DOCTYPE html> <html> <head> <meta charset="UTF-8"> <script src="kiko-uploader.min.js"></script> </head> <body> <kiko-uploader max-width="200" max-height="280" show-filename="true" show-filesize="true" show-file-list="true" groups='[{"id":"docs","name":"Documents","order":0}]' ></kiko-uploader> </body> </html>

JavaScript (optional)

const uploader = document.querySelector('kiko-uploader'); // Get uploaded files const files = uploader.getUploadedFiles(); console.log('Uploaded files:', files); // Listen to events uploader.addEventListener('file-added', (event) => { console.log('File added:', event.detail); console.log('All files:', uploader.getUploadedFiles()); }); // Add a group programmatically uploader.addGroup('New Group');

Set Files Programmatically (optional)

// Set files programmatically (useful for loading saved state) const sampleFiles = [ { file: new File(['sample content'], 'document1.pdf', { type: 'application/pdf' }), group: 'Documents', description: 'Important document' }, { file: new File(['sample content'], 'image1.jpg', { type: 'image/jpeg' }), group: 'Images', description: 'Sample image' }, { file: new File(['sample content'], 'report.pdf', { type: 'application/pdf' }), group: 'Reports', description: 'Monthly report' } ]; await uploader.setUploadedFiles(sampleFiles);

React

Use React refs to access the component's methods and events.

React Component

import React, { useRef, useEffect } from 'react'; import './kiko-uploader.min.js'; function KikoUploaderApp() { const uploaderRef = useRef(null); useEffect(() => { const uploader = uploaderRef.current; const handleFileAdded = (event) => { console.log('File added:', event.detail); }; uploader.addEventListener('file-added', handleFileAdded); return () => { uploader.removeEventListener('file-added', handleFileAdded); }; }, []); const getFiles = () => { const files = uploaderRef.current.getUploadedFiles(); console.log('Uploaded files:', files); }; const addGroup = () => { uploaderRef.current.addGroup('New Group'); }; return ( <div> <kiko-uploader ref={uploaderRef} max-width="200" max-height="280" show-filename="true" groups='[{"id":"docs","name":"Documents"}]' /> <button onClick={getFiles}>Get Files</button> <button onClick={addGroup}>Add Group</button> </div> ); } export default KikoUploaderApp;

React with Redux

Integrate the file uploader with Redux for centralized state management.

Redux Store Setup

// store.js import { createStore } from 'redux'; import kikoUploaderReducer from './reducers/kikoUploaderReducer'; const store = createStore(kikoUploaderReducer); export default store;

Actions

// actions/kikoUploaderActions.js export const ADD_FILE = 'ADD_FILE'; export const REMOVE_FILE = 'REMOVE_FILE'; export const SET_FILES = 'SET_FILES'; export const UPDATE_DESCRIPTION = 'UPDATE_DESCRIPTION'; export const ADD_GROUP = 'ADD_GROUP'; export const CLEAR_FILES = 'CLEAR_FILES'; export const addFile = (fileData) => ({ type: ADD_FILE, payload: fileData }); export const removeFile = (fileIndex) => ({ type: REMOVE_FILE, payload: fileIndex }); export const setFiles = (files) => ({ type: SET_FILES, payload: files }); export const updateDescription = (fileIndex, description) => ({ type: UPDATE_DESCRIPTION, payload: { fileIndex, description } }); export const addGroup = (groupName) => ({ type: ADD_GROUP, payload: groupName }); export const clearFiles = () => ({ type: CLEAR_FILES });

Reducer

// reducers/kikoUploaderReducer.js import { ADD_FILE, REMOVE_FILE, SET_FILES, UPDATE_DESCRIPTION, ADD_GROUP, CLEAR_FILES } from '../actions/kikoUploaderActions'; const initialState = { files: [], groups: [ { id: 'reports', name: 'Annual Reports', order: 0 }, { id: 'documents', name: 'General documents', order: 1 }, { id: 'images', name: 'Images', order: 2 } ] }; const kikoUploaderReducer = (state = initialState, action) => { switch (action.type) { case ADD_FILE: return { ...state, files: [...state.files, action.payload] }; case REMOVE_FILE: return { ...state, files: state.files.filter((_, index) => index !== action.payload) }; case SET_FILES: return { ...state, files: action.payload }; case UPDATE_DESCRIPTION: return { ...state, files: state.files.map((file, index) => index === action.payload.fileIndex ? { ...file, description: action.payload.description } : file ) }; case ADD_GROUP: const newGroupId = `group_${Date.now()}`; return { ...state, groups: [...state.groups, { id: newGroupId, name: action.payload, order: state.groups.length }] }; case CLEAR_FILES: return { ...state, files: [] }; default: return state; } }; export default kikoUploaderReducer;

React Component with Redux

// components/KikoUploaderRedux.js import React, { useRef, useEffect } from 'react'; import { useSelector, useDispatch } from 'react-redux'; import { addFile, removeFile, setFiles, updateDescription, addGroup, clearFiles } from '../actions/kikoUploaderActions'; import './kiko-uploader.min.js'; function KikoUploaderRedux() { const uploaderRef = useRef(null); const dispatch = useDispatch(); const { files, groups } = useSelector(state => state); useEffect(() => { const uploader = uploaderRef.current; const handleFileAdded = (event) => { dispatch(addFile(event.detail)); }; const handleFileRemoved = (event) => { dispatch(removeFile(event.detail.index)); }; const handleDescriptionUpdated = (event) => { dispatch(updateDescription(event.detail.fileIndex, event.detail.description)); }; const handleFilesCleared = () => { dispatch(clearFiles()); }; uploader.addEventListener('file-added', handleFileAdded); uploader.addEventListener('file-removed', handleFileRemoved); uploader.addEventListener('description-updated', handleDescriptionUpdated); uploader.addEventListener('files-cleared', handleFilesCleared); return () => { uploader.removeEventListener('file-added', handleFileAdded); uploader.removeEventListener('file-removed', handleFileRemoved); uploader.removeEventListener('description-updated', handleDescriptionUpdated); uploader.removeEventListener('files-cleared', handleFilesCleared); }; }, [dispatch]); const handleGetFiles = () => { const uploaderFiles = uploaderRef.current.getUploadedFiles(); dispatch(setFiles(uploaderFiles)); console.log('Files from Redux state:', files); }; const handleAddGroup = () => { const groupName = 'New Redux Group'; dispatch(addGroup(groupName)); uploaderRef.current.addGroup(groupName); }; const handleClearFiles = () => { uploaderRef.current.clearAllFiles(); }; return ( <div> <h3>Redux State</h3> <p>Files in Redux: {files.length}</p> <p>Groups in Redux: {groups.length}</p> <div style={{ marginBottom: '20px' }}> <button onClick={handleGetFiles} style={{ marginRight: '10px' }}> Sync Files to Redux </button> <button onClick={handleAddGroup} style={{ marginRight: '10px' }}> Add Group </button> <button onClick={handleClearFiles}> Clear All Files </button> </div> <kiko-uploader ref={uploaderRef} max-width="200" max-height="280" show-filename="true" show-filesize="true" groups={JSON.stringify(groups)} /> </div> ); } export default KikoUploaderRedux;

App Setup

// App.js import React from 'react'; import { Provider } from 'react-redux'; import store from './store'; import KikoUploaderRedux from './components/KikoUploaderRedux'; function App() { return ( <Provider store={store}> <div className="App"> <h1>Kiko Uploader with Redux</h1> <KikoUploaderRedux /> </div> </Provider> ); } export default App;

Vue.js

Use Vue's ref system to access the component.

Vue Component

<template> <div> <kiko-uploader ref="uploader" max-width="200" max-height="280" show-filename="true" @file-added="onFileAdded" /> <button @click="getFiles">Get Files</button> <button @click="addGroup">Add Group</button> </div> </template> <script> import './kiko-uploader.min.js'; export default { mounted() { // Component is ready }, methods: { onFileAdded(event) { console.log('File added:', event.detail); }, getFiles() { const files = this.$refs.uploader.getUploadedFiles(); console.log('Files:', files); }, addGroup() { this.$refs.uploader.addGroup('New Group'); } } } </script>

Angular

Use ViewChild to access the native element.

Angular Component

import { Component, ViewChild, ElementRef } from '@angular/core'; @Component({ selector: 'app-file-uploader', template: ` <kiko-uploader #uploader max-width="200" max-height="280" show-filename="true" (file-added)="onFileAdded($event)" ></kiko-uploader> <button (click)="getFiles()">Get Files</button> <button (click)="addGroup()">Add Group</button> ` }) export class KikoUploaderComponent { @ViewChild('uploader') uploader!: ElementRef; ngAfterViewInit() { // Component is ready } onFileAdded(event: any) { console.log('File added:', event.detail); } getFiles() { const files = this.uploader.nativeElement.getUploadedFiles(); console.log('Files:', files); } addGroup() { this.uploader.nativeElement.addGroup('New Group'); } }
