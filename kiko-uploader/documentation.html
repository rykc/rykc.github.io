Methods

getUploadedFiles()

Returns uploaded files with metadata, server IDs, and deleted file IDs for sync.

const result = uploader.getUploadedFiles()
// Returns: { 
//   files: [{ 
//     id,           // Server ID (>0) or -1 for new/local files
//     file,         // File object (Blob/File) or placeholder for server files
//     fileName,     // Display name (editable)
//     order,        // Overall file order
//     group,        // Group name
//     groupId,      // Group ID
//     groupItemOrder, // Order within group
//     thumbnail,    // Base64 thumbnail or URL
//     description1, // First description field
//     description2, // Second description field
//     size,         // File size in bytes
//     fileURL       // Server file URL (for server-backed files)
//   }], 
//   deleted: [id,...] // Array of deleted server file IDs
// }

setUploadedFiles(files)

Programmatically set files in the uploader, including server-backed files with URLs and metadata.

await uploader.setUploadedFiles([{
  file?,          // File object (optional if fileURL provided)
  group?,         // Group name or ID
  groupId?,       // Specific group ID
  thumbnail?,     // Base64 thumbnail or thumbnail URL
  thumbnailURL?,  // URL to thumbnail image
  description1?,  // First description
  description2?,  // Second description
  fileName?,      // Display file name
  id?,            // Server file ID (>0) or -1 for local
  fileURL?,       // Server file URL (for server-backed files)
  size?           // File size in bytes
}])
// Returns a Promise

setGroups(groups)

Set the groups configuration programmatically with full support for nested structures.

uploader.setGroups([{
  id,              // Unique identifier
  name,            // Display name
  order?,          // Sort order (default: by array position)
  width?,          // CSS width (e.g., "50%", "300px")
  "has-file"?,     // Can hold files (default: 1/true)
  groups?,         // Array of nested sub-groups
  "single-folder-mode"?, // Enforce single folder for children
  "single-folder-hint"?, // Custom hint for this group
  "tab-mode"?,     // Display sub-groups as tabs
  "default-expand"? // Expanded by default (default: true)
}])

Events

ready

Fired when component is fully initialized and ready for use.

file-added

Fired when a file is uploaded. Detail: {file, index}

file-removed

Fired when a file is removed. Detail: {index}

file-moved

Fired when a file is moved between groups. Detail: {fileIndex, groupId, fromGroupId}

description-updated

Fired when file description or name is updated. Detail: {fileIndex, description: {desc1, desc2}, fileName}

files-cleared

Fired when all files are cleared.

files-set

Fired when files are set programmatically. Detail: {count}

dirty

Fired when file state changes (for tracking unsaved changes). Detail: {isDirty, triggeringEvent}

Others

Kiko Uploader supports server-backed files with comprehensive metadata tracking:

Loading Server Files

Set files from server with file IDs, URLs, and thumbnails:

await uploader.setUploadedFiles([
  {
    id: 123,                              // Server file ID (>0)
    fileName: "document.pdf",             // Display name
    fileURL: "https://example.com/files/document.pdf",
    thumbnailURL: "https://example.com/thumbnails/document.jpg",
    size: 2048576,                        // File size in bytes
    group: "Documents",                   // Group name or ID
    groupId: "doc-folder",                // Specific group ID
    description1: "Annual Report",        // First description
    description2: "Financial year 2024"   // Second description
  },
  // ... more files
])

Tracking Deletions

When calling getUploadedFiles(), deleted server files are tracked:

const result = uploader.getUploadedFiles()
// result.files: current files
// result.deleted: array of deleted server file IDs
// Use result.deleted to remove files from server

// After sync, mark as clean
uploader.markAsClean()

Mixed Local & Server Files

Files with id === -1 are new/local files. Files with id > 0 are server-backed:

const result = uploader.getUploadedFiles()
result.files.forEach(file => {
  if (file.id === -1) {
    // New local file - upload to server
    uploadToServer(file.file, file.fileName)
  } else {
    // Server-backed file - update metadata if changed
    updateMetadata(file.id, file.description1, file.description2)
  }
})

ZIP Export with Server Files

The built-in ZIP export automatically fetches server files and includes them in the download with proper folder structure preserved.

React with Redux

Integrate the file uploader with Redux for centralized state management.

Redux Store Setup

// store.js import { createStore } from 'redux'; import kikoUploaderReducer from './reducers/kikoUploaderReducer'; const store = createStore(kikoUploaderReducer); export default store;

Actions

// actions/kikoUploaderActions.js export const ADD_FILE = 'ADD_FILE'; export const REMOVE_FILE = 'REMOVE_FILE'; export const SET_FILES = 'SET_FILES'; export const UPDATE_DESCRIPTION = 'UPDATE_DESCRIPTION'; export const ADD_GROUP = 'ADD_GROUP'; export const CLEAR_FILES = 'CLEAR_FILES'; export const SET_SIZE_LIMITS = 'SET_SIZE_LIMITS'; export const TOGGLE_CARD_VIEW = 'TOGGLE_CARD_VIEW'; export const SET_DIRTY = 'SET_DIRTY'; export const addFile = (fileData) => ({ type: ADD_FILE, payload: fileData }); export const removeFile = (fileIndex) => ({ type: REMOVE_FILE, payload: fileIndex }); export const setFiles = (files) => ({ type: SET_FILES, payload: files }); export const updateDescription = (fileIndex, description) => ({ type: UPDATE_DESCRIPTION, payload: { fileIndex, description } }); export const addGroup = (groupName) => ({ type: ADD_GROUP, payload: groupName }); export const clearFiles = () => ({ type: CLEAR_FILES }); export const setSizeLimits = (maxFileSize, maxTotalSize) => ({ type: SET_SIZE_LIMITS, payload: { maxFileSize, maxTotalSize } }); export const toggleCardView = () => ({ type: TOGGLE_CARD_VIEW }); export const setDirty = (isDirty) => ({ type: SET_DIRTY, payload: isDirty });

Reducer

// reducers/kikoUploaderReducer.js import { ADD_FILE, REMOVE_FILE, SET_FILES, UPDATE_DESCRIPTION, ADD_GROUP, CLEAR_FILES, SET_SIZE_LIMITS, TOGGLE_CARD_VIEW, SET_DIRTY } from '../actions/kikoUploaderActions'; const initialState = { files: [], groups: [ { id: 'reports', name: 'Annual Reports', order: 0 }, { id: 'documents', name: 'General documents', order: 1 }, { id: 'images', name: 'Images', order: 2 } ], maxFileSize: 20, maxTotalSize: 100, cardView: false, isDirty: false }; const kikoUploaderReducer = (state = initialState, action) => { switch (action.type) { case ADD_FILE: return { ...state, files: [...state.files, action.payload] }; case REMOVE_FILE: return { ...state, files: state.files.filter((_, index) => index !== action.payload) }; case SET_FILES: return { ...state, files: action.payload }; case UPDATE_DESCRIPTION: return { ...state, files: state.files.map((file, index) => index === action.payload.fileIndex ? { ...file, description: action.payload.description } : file ) }; case ADD_GROUP: const newGroupId = `group_${Date.now()}`; return { ...state, groups: [...state.groups, { id: newGroupId, name: action.payload, order: state.groups.length }] }; case CLEAR_FILES: return { ...state, files: [] }; case SET_SIZE_LIMITS: return { ...state, maxFileSize: action.payload.maxFileSize, maxTotalSize: action.payload.maxTotalSize }; case TOGGLE_CARD_VIEW: return { ...state, cardView: !state.cardView }; case SET_DIRTY: return { ...state, isDirty: action.payload }; default: return state; } }; default: return state; } }; export default kikoUploaderReducer;
